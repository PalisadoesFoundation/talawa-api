export const Details = ({
  dataOpen,
  dataClose,
  children,
  startOpen = false,
}) => {
  const [open, setOpen] = useState(startOpen);
  return (
    <details
      {...(open ? { open: true } : {})}
      className="details"
      style={{
        border: "none",
        boxShadow: "none",

      </summary>
      {open && children}

  );
